数组

every()：

对数组中的每一项运行给定函数，如果该函数对每一项都返回true，则返回true。


var a = [1,2,3]

console.log(a.every(function(item) { return item > 0}))
console.log(a.every(function(item) { return item > 1}))

// 输出
 true
 false

filter()：

对数组中的每一项运行给定函数，返回该函数会返回true的项组成的数组。


var a = [1,2,3]
console.log(a.filter(function(item) { return item > 0}))

console.log(a.filter(function(item) { return item > 1}))

// 输出
[1, 2, 3]
[2, 3]

forEach()：

对数组中的每一项运行给定函数。这个方法没有返回值。

var a = [1,2,3]
console.log(a.forEach(function(item) { console.log(item)}))
// 输出
 1
 2
 3

map()：

对数组中的每一项运行给定函数，返回每次函数调用的结果组成的数组。

var a = [1,2,3]
console.log(a.map(function(item) { return item + 3}))
// 输出
 [4, 5, 6]

some()：

对数组中的每一项运行给定函数，如果该函数对任一项返回true，则返回true。

var a = [1,2,3]
console.log(a.some(function(item) { return item >=3}))
console.log(a.some(function(item) { return item >=4}))
// 输出
true
false

向下取整

// 向下取整,丢弃小数部分
Math.floor(5/2)　　// 2

// 四舍五入
